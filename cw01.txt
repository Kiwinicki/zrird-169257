-- 1.

CREATE TABLE regions (
    region_id NUMBER CONSTRAINT reg_id_pk PRIMARY KEY,
    region_name VARCHAR2(25)
);

CREATE TABLE countries (
    country_id CHAR(2) CONSTRAINT country_c_id_pk PRIMARY KEY,
    country_name VARCHAR2(40),
    region_id NUMBER CONSTRAINT countr_reg_fk REFERENCES regions(region_id)
);

CREATE TABLE locations (
    location_id NUMBER CONSTRAINT loc_id_pk PRIMARY KEY,
    street_address VARCHAR2(40),
    postal_code VARCHAR2(12),
    city VARCHAR2(30),
    country_id CHAR(2) CONSTRAINT loc_c_id_fk REFERENCES countries(country_id)
);

CREATE TABLE jobs (
    job_id VARCHAR2(10) CONSTRAINT job_id_pk PRIMARY KEY,
    job_title VARCHAR2(35) NOT NULL,
    min_salary NUMBER(8),
    max_salary NUMBER(8)
);

CREATE TABLE departments (
    department_id NUMBER CONSTRAINT dept_id_pk PRIMARY KEY,
    department_name VARCHAR2(30) NOT NULL,
    manager_id NUMBER,
    location_id NUMBER CONSTRAINT dept_loc_fk REFERENCES locations(location_id)
);

CREATE TABLE employees (
    employee_id NUMBER CONSTRAINT emp_emp_id_pk PRIMARY KEY,
    first_name VARCHAR2(20),
    last_name VARCHAR2(25) NOT NULL,
    email VARCHAR2(25) NOT NULL UNIQUE,
    hire_date DATE NOT NULL,
    job_id VARCHAR2(10) CONSTRAINT emp_job_fk REFERENCES jobs(job_id),
    salary NUMBER(8,2),
    commission_pct NUMBER(2,2),
    manager_id NUMBER CONSTRAINT emp_manager_fk REFERENCES employees(employee_id),
    department_id NUMBER
);

CREATE TABLE job_history (
    employee_id NUMBER CONSTRAINT jhist_emp_fk REFERENCES employees(employee_id),
    start_date DATE,
    end_date DATE,
    job_id VARCHAR2(10) CONSTRAINT jhist_job_fk REFERENCES jobs(job_id),
    department_id NUMBER CONSTRAINT jhist_dept_fk REFERENCES departments(department_id),
    CONSTRAINT jhist_emp_id_st_date_pk PRIMARY KEY (employee_id, start_date)
);

-- 2.

ALTER TABLE locations ADD (state_province VARCHAR2(25));
ALTER TABLE employees ADD (phone_number VARCHAR2(20));

ALTER TABLE departments ADD CONSTRAINT dept_mgr_fk FOREIGN KEY (manager_id) REFERENCES employees (employee_id);
ALTER TABLE employees ADD CONSTRAINT emp_dept_fk FOREIGN KEY (department_id) REFERENCES departments (department_id);

ALTER TABLE jobs ADD CONSTRAINT job_salary_chk CHECK (max_salary - min_salary >= 2000);

DROP TABLE job_history;
