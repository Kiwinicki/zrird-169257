-- zad1: ranking pracowników wg pensji (remisy mają ten sam numer)
CREATE OR REPLACE VIEW v_cw03_zad1 AS
SELECT employee_id,
       first_name,
       last_name,
       salary,
       RANK() OVER (ORDER BY salary DESC) AS ranking_pensji
FROM employees;

-- zad2: całkowita suma pensji bez grupowania
CREATE OR REPLACE VIEW v_cw03_zad2 AS
SELECT employee_id,
       first_name,
       last_name,
       salary,
       SUM(salary) OVER () AS suma_pensji_wszyscy
FROM employees;

-- zad3: nazwisko, produkt, skumulowana wartość sprzedaży pracownika + ranking wartości zamówienia globalnie
CREATE OR REPLACE VIEW v_cw03_zad3 AS
SELECT e.last_name,
       p.product_name AS nazwa_produktu,
       SUM(s.price * s.quantity)
         OVER (PARTITION BY s.employee_id ORDER BY s.sale_date, s.sale_id
               ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW) AS skumulowana_wartosc_sprzedazy_pracownika,
       RANK() OVER (ORDER BY (s.price * s.quantity) DESC) AS ranking_wartosci_zamowienia
FROM sales s
JOIN employees e ON e.employee_id = s.employee_id
JOIN products p ON p.product_id = s.product_id;

-- zad4: dla każdego wiersza SALES: pracownik, produkt, cena, liczba transakcji produktu danego dnia,
--       suma zapłacona danego dnia za produkt, poprzednia i kolejna cena produktu
CREATE OR REPLACE VIEW v_cw03_zad4 AS
SELECT e.last_name,
       p.product_name AS nazwa_produktu,
       s.price AS cena_produktu,
       COUNT(*) OVER (
         PARTITION BY TRUNC(s.sale_date), s.product_id
       ) AS liczba_transakcji_produktu_tego_dnia,
       SUM(s.price * s.quantity) OVER (
         PARTITION BY TRUNC(s.sale_date), s.product_id
       ) AS suma_zaplacona_tego_dnia_za_produkt,
       LAG(s.price) OVER (
         PARTITION BY s.product_id ORDER BY s.sale_date, s.sale_id
       ) AS poprzednia_cena,
       LEAD(s.price) OVER (
         PARTITION BY s.product_id ORDER BY s.sale_date, s.sale_id
       ) AS kolejna_cena
FROM sales s
JOIN employees e ON e.employee_id = s.employee_id
JOIN products p ON p.product_id = s.product_id;

-- zad5: nazwa produktu, cena, suma całkowita w miesiącu + suma rosnąca w miesiącu dla produktu
CREATE OR REPLACE VIEW v_cw03_zad5 AS
SELECT p.product_name AS nazwa_produktu,
       s.price AS cena_produktu,
       TRUNC(s.sale_date, 'MM') AS miesiac,
       SUM(s.price * s.quantity) OVER (
         PARTITION BY s.product_id, TRUNC(s.sale_date, 'MM')
       ) AS suma_miesieczna,
       SUM(s.price * s.quantity) OVER (
         PARTITION BY s.product_id, TRUNC(s.sale_date, 'MM')
         ORDER BY s.sale_date, s.sale_id
         ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
       ) AS suma_rosnaca_w_miesiacu
FROM sales s
JOIN products p ON p.product_id = s.product_id;

-- zad6: cena produktu z 2022 i 2023 z tego samego dnia (MM-DD) + różnica + nazwa i kategoria
-- (w HR nie ma historii cen w PRODUCTS, więc bierzemy ceny z SALES; w razie wielu transakcji danego dnia używamy AVG)
CREATE OR REPLACE VIEW v_cw03_zad6 AS
WITH dzienne_ceny AS (
  SELECT s.product_id AS product_id,
         TRUNC(s.sale_date) AS dzien,
         AVG(s.price) AS cena
  FROM sales s
  GROUP BY s.product_id, TRUNC(s.sale_date)
),
c2022 AS (
  SELECT product_id, dzien, cena
  FROM dzienne_ceny
  WHERE EXTRACT(YEAR FROM dzien) = 2022
),
c2023 AS (
  SELECT product_id, dzien, cena
  FROM dzienne_ceny
  WHERE EXTRACT(YEAR FROM dzien) = 2023
)
SELECT p.product_name AS nazwa_produktu,
       p.product_category AS kategoria,
       TO_CHAR(c2022.dzien, 'MM-DD') AS dzien,
       c2022.cena AS cena_2022,
       c2023.cena AS cena_2023,
       (c2023.cena - c2022.cena) AS roznica_cen
FROM c2022
JOIN c2023
  ON c2023.product_id = c2022.product_id
 AND TO_CHAR(c2023.dzien, 'MM-DD') = TO_CHAR(c2022.dzien, 'MM-DD')
JOIN products p ON p.product_id = c2022.product_id;

-- zad7: kategoria, produkt, cena, min/max w kategorii + różnica
CREATE OR REPLACE VIEW v_cw03_zad7 AS
SELECT p.product_category AS kategoria,
       p.product_name AS nazwa_produktu,
       s.price AS cena,
       MIN(s.price) OVER (PARTITION BY p.product_category) AS min_cena_w_kategorii,
       MAX(s.price) OVER (PARTITION BY p.product_category) AS max_cena_w_kategorii,
       ( MAX(s.price) OVER (PARTITION BY p.product_category)
       - MIN(s.price) OVER (PARTITION BY p.product_category) ) AS roznica_max_min
FROM sales s
JOIN products p ON p.product_id = s.product_id;

-- zad8: średnia krocząca (poprzednia, bieżąca, następna) ceny produktu wg dat
CREATE OR REPLACE VIEW v_cw03_zad8 AS
SELECT p.product_name AS nazwa_produktu,
       s.sale_date AS data,
       s.price AS cena,
       AVG(s.price) OVER (
         PARTITION BY s.product_id
         ORDER BY s.sale_date, s.sale_id
         ROWS BETWEEN 1 PRECEDING AND 1 FOLLOWING
       ) AS srednia_kroczaca_3
FROM sales s
JOIN products p ON p.product_id = s.product_id;

-- zad9: nazwa produktu, kategoria + rank/row_number/dense_rank cen w kategorii
CREATE OR REPLACE VIEW v_cw03_zad9 AS
SELECT p.product_name AS nazwa_produktu,
       p.product_category AS kategoria,
       s.price AS cena,
       RANK() OVER (PARTITION BY p.product_category ORDER BY s.price DESC) AS ranking_cen,
       ROW_NUMBER() OVER (PARTITION BY p.product_category ORDER BY s.price DESC) AS numer_wiersza_w_kategorii,
       DENSE_RANK() OVER (PARTITION BY p.product_category ORDER BY s.price DESC) AS dense_ranking_cen
FROM sales s
JOIN products p ON p.product_id = s.product_id;

-- zad10: dla każdego wiersza SALES: pracownik, produkt, rosnąca wartość sprzedaży pracownika wg dat + ranking zamówienia globalnie
CREATE OR REPLACE VIEW v_cw03_zad10 AS
SELECT e.last_name,
       p.product_name AS nazwa_produktu,
       SUM(s.price * s.quantity)
         OVER (PARTITION BY s.employee_id ORDER BY s.sale_date, s.sale_id
               ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW) AS wartosc_rosnaca_pracownika,
       RANK() OVER (ORDER BY (s.price * s.quantity) DESC) AS ranking_wartosci_zamowienia_globalnie
FROM sales s
JOIN employees e ON e.employee_id = s.employee_id
JOIN products p ON p.product_id = s.product_id;

-- zad11: bez funkcji okienkowych: pracownicy (imię+nazwisko) i stanowisko, którzy uczestniczyli w sprzedaży
CREATE OR REPLACE VIEW v_cw03_zad11 AS
SELECT DISTINCT
       e.first_name || ' ' || e.last_name AS imie_nazwisko,
       j.job_title
FROM sales s
JOIN employees e ON e.employee_id = s.employee_id
JOIN jobs j ON j.job_id = e.job_id;
